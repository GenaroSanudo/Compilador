func int fibo_recur(int n){
    if (n <= 1){
        return (n);
    }else{
        return(fibo_recur(n-1); + fibo_recur(n-2););
    };
};

func int fibo_iterative(int n){
    var int a, b, c;

    a = 0;
    b = 1;

    write(a,b);
    while(n-2 >= 0){
        c=a+b;
        a = b;
        b = c;
        write(c);
        n=n-1;
    };
    return (c);
};

main(){
var int i, num;
num = 10;

write("Fibonaci number: ", fibo_recur(10););

write("Fibonacci secuence: ");
for (i = 0 to num+1){
    write(fibo_recur(i););
}; 
write ("Fibonacci iterative: ");
write(fibo_iterative(num););
};
