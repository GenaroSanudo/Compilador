var int mat[2][1];
var int mat2[1][3];
var int result[2][3];


func void mat_mult(int length, int length2, int length3){
    var int i, j, k;

    for (i = 0 to length){
        for (j = 0 to length2){
            for (k = 0 to length3){
                result[i][j] = result[i][j] + mat[i][k] * mat2[k][j];
            };
        };
    };


    };
